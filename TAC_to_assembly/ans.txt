.text
main:
addi $sp $sp -4
sw $ra ($sp)
addi $sp $sp -4
sw $fp ($sp)
move $fp $sp
jal _main
lw $fp ($sp)
addi $sp $sp 4
lw $ra ($sp)
addi $sp $sp 4
li $v0 10
syscall
_main:
addiu $sp $sp -48
li $v0 5 
syscall 
sw $v0 -4($fp)
lw $t0 -4($fp)
bne $t0 0 OutputBoolIsTrue
la $a0 false
j OutputBoolContinue
OutputBoolIsTrue:
la $a0 true
OutputBoolContinue:
li $v0 4
syscall 
addiu $sp $sp 48
jr $ra 
.data
newline: .asciiz "\n"
true: .asciiz "true"
false: .asciiz "false"
