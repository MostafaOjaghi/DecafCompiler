.text
main:
addi $sp $sp -4
sw $ra ($sp)
addi $sp $sp -4
sw $fp ($sp)
move $fp $sp
jal func_main
lw $fp ($sp)
addi $sp $sp 4
lw $ra ($sp)
addi $sp $sp 4
li $v0 10
syscall
func_f:
addiu $sp $sp -40
li.s $f0 0.0
s.s $f0 -4($fp)
li.s $f0 0.0
s.s $f0 -8($fp)
li $t0 0
sw $t0 -12($fp)
li $t0 0
sw $t0 -16($fp)
li.s $f0 0.0
s.s $f0 -20($fp)
li $t0 10
sw $t0 -24($fp)
lw $t0 -24($fp)
sw $t0 -12($fp)
li $t0 20
sw $t0 -28($fp)
lw $t0 -28($fp)
sw $t0 -16($fp)
lw $t0 -12($fp)
lw $t1 -16($fp)
add $t0 $t0 $t1
sw $t0 -32($fp)
l.s $f0 -32($fp)
s.s $f0 -20($fp)
li $v0 2 
l.s $f12 -20($fp)
syscall 
li $v0 4
la $a0 newline
syscall
li.s $f0 3.1
s.s $f0 -36($fp)
l.s $f0 -36($fp)
s.s $f0 -4($fp)
l.s $f0 8($fp)
l.s $f1 -4($fp)
add.s $f0 $f0 $f1
s.s $f0 -40($fp)
l.s $f0 -40($fp)
s.s $f0 -8($fp)
li $v0 2 
l.s $f12 8($fp)
syscall 
li $v0 4
la $a0 newline
syscall
li $v0 2 
l.s $f12 -4($fp)
syscall 
li $v0 4
la $a0 newline
syscall
li $v0 2 
l.s $f12 -8($fp)
syscall 
li $v0 4
la $a0 newline
syscall
addiu $sp $sp 40
jr $ra 
func_main:
addiu $sp $sp -8
li.s $f0 2.3
s.s $f0 -4($fp)
addi $sp $sp -4
l.s $f0 -4($fp)
s.s $f0 0($sp)
addi $sp $sp -4
sw $ra ($sp)
addi $sp $sp -4
sw $fp ($sp)
move $fp $sp
jal func_f
lw $fp ($sp)
addi $sp $sp 4
lw $ra ($sp)
addi $sp $sp 4
sw $v0 -8($fp)
addi $sp $sp 4
addiu $sp $sp 8
jr $ra
